// src/util/db.ts
import Database from "better-sqlite3";
import { join } from "path";
import { existsSync } from "fs";

const SCHEMA_VERSION = 1;

export function openDb(historyDir: string): Database.Database {
  const dbPath = join(historyDir, "history.db");
  const db = new Database(dbPath);
  db.pragma("journal_mode = WAL");
  db.pragma("busy_timeout = 5000");
  db.pragma("foreign_keys = OFF");
  initSchema(db);
  return db;
}

export function hasDb(historyDir: string): boolean {
  return existsSync(join(historyDir, "history.db"));
}

import { execSync } from "child_process";

export interface GitCommit {
  hash: string;
  message: string;
  author: string;
  email: string;
  timestamp: string;
  branch: string;
  parentHash: string;
  mergeCommit: boolean;
  insertions: number;
  deletions: number;
  filesChanged: number;
}

export interface GitFile {
  path: string;
  status: string; // A, M, D, R
  insertions: number;
  deletions: number;
  oldPath: string | null;
}

export interface GitBranch {
  name: string;
  headHash: string;
  isActive: boolean;
}

export interface GitTag {
  name: string;
  hash: string;
  timestamp: string;
  message: string;
}

interface LogOptions {
  limit?: number;
  since?: string;
  until?: string;
  author?: string;
  branch?: string;
}

function run(cwd: string, cmd: string): string {
  try {
    return execSync(cmd, { cwd, encoding: "utf-8", timeout: 30000 }).trim();
  } catch {
    return "";
  }
}

export function isGitRepo(dir: string): boolean {
  try {
    execSync("git rev-parse --git-dir", { cwd: dir, encoding: "utf-8", stdio: "pipe" });
    return true;
  } catch {
    return false;
  }
}

export function getLastCommitHash(cwd: string): string {
  return run(cwd, "git rev-parse HEAD");
}

export function getLog(cwd: string, opts: LogOptions = {}): GitCommit[] {
  const SEP = "---GIT-SKILL-SEP---";
  const format = ["%H", "%s", "%an", "%ae", "%aI", "%D", "%P"].join(SEP);
  let cmd = `git log --format="${format}" --numstat`;

  if (opts.limit) cmd += ` -n ${opts.limit}`;
  if (opts.since) cmd += ` --since="${opts.since}"`;
  if (opts.until) cmd += ` --until="${opts.until}"`;
  if (opts.author) cmd += ` --author="${opts.author}"`;
  if (opts.branch) cmd += ` ${opts.branch}`;

  const output = run(cwd, cmd);
  if (!output) return [];

  const commits: GitCommit[] = [];
  const lines = output.split("\n");
  let current: GitCommit | null = null;
  let insertions = 0, deletions = 0, filesChanged = 0;

  for (const line of lines) {
    if (line.includes(SEP)) {
      if (current) {
        current.insertions = insertions;
        current.deletions = deletions;
        current.filesChanged = filesChanged;
        commits.push(current);
      }
      const parts = line.split(SEP);
      const parents = parts[6] || "";
      current = {
        hash: parts[0],
        message: parts[1],
        author: parts[2],
        email: parts[3],
        timestamp: parts[4],
        branch: extractBranch(parts[5]),
        parentHash: parents,
        mergeCommit: parents.split(" ").length > 1,
        insertions: 0,
        deletions: 0,
        filesChanged: 0,
      };
      insertions = 0;
      deletions = 0;
      filesChanged = 0;
    } else if (line.trim() && current) {
      const match = line.match(/^(\d+|-)\t(\d+|-)\t(.+)/);
      if (match) {
        const ins = match[1] === "-" ? 0 : parseInt(match[1], 10);
        const del = match[2] === "-" ? 0 : parseInt(match[2], 10);
        insertions += ins;
        deletions += del;
        filesChanged++;
      }
    }
  }

  if (current) {
    current.insertions = insertions;
    current.deletions = deletions;
    current.filesChanged = filesChanged;
    commits.push(current);
  }

  return commits;
}

function extractBranch(refStr: string): string {
  if (!refStr) return "";
  const parts = refStr.split(",").map(s => s.trim());
  for (const p of parts) {
    if (p.startsWith("HEAD -> ")) return p.replace("HEAD -> ", "");
    if (!p.includes("tag:") && !p.includes("HEAD")) return p;
  }
  return "";
}

export function getDiffTree(cwd: string, hash: string): GitFile[] {
  const output = run(cwd, `git diff-tree --no-commit-id -r --numstat -M ${hash}`);
  if (!output) return [];

  const files: GitFile[] = [];
  const numstatLines = output.split("\n");

  // Also get status letters
  const statusOutput = run(cwd, `git diff-tree --no-commit-id -r --name-status -M ${hash}`);
  const statusMap = new Map<string, { status: string; oldPath: string | null }>();

  if (statusOutput) {
    for (const line of statusOutput.split("\n")) {
      const parts = line.split("\t");
      if (parts.length >= 2) {
        const status = parts[0][0]; // R100 -> R
        const filePath = parts.length === 3 ? parts[2] : parts[1];
        const oldPath = parts.length === 3 ? parts[1] : null;
        statusMap.set(filePath, { status, oldPath });
      }
    }
  }

  for (const line of numstatLines) {
    const match = line.match(/^(\d+|-)\t(\d+|-)\t(.+)/);
    if (match) {
      let filePath = match[3];
      // Handle renames: old => new
      if (filePath.includes(" => ")) {
        const renameParts = filePath.split(" => ");
        filePath = renameParts[1];
      }
      const info = statusMap.get(filePath) || { status: "M", oldPath: null };
      files.push({
        path: filePath,
        status: info.status,
        insertions: match[1] === "-" ? 0 : parseInt(match[1], 10),
        deletions: match[2] === "-" ? 0 : parseInt(match[2], 10),
        oldPath: info.oldPath,
      });
    }
  }

  return files;
}

export function getBranches(cwd: string): GitBranch[] {
  const output = run(cwd, "git branch -a --format='%(refname:short)\t%(objectname:short)\t%(HEAD)'");
  if (!output) return [];

  return output.split("\n").filter(Boolean).map(line => {
    const [name, hash, head] = line.split("\t");
    return {
      name: name.replace(/^'|'$/g, ""),
      headHash: hash,
      isActive: head?.trim() === "*",
    };
  });
}

export function getTags(cwd: string): GitTag[] {
  const output = run(cwd, "git tag -l --format='%(refname:short)\t%(objectname:short)\t%(creatordate:iso-strict)\t%(subject)'");
  if (!output) return [];

  return output.split("\n").filter(Boolean).map(line => {
    const parts = line.split("\t");
    return {
      name: parts[0]?.replace(/^'|'$/g, "") || "",
      hash: parts[1] || "",
      timestamp: parts[2] || "",
      message: parts[3] || "",
    };
  });
}

import { readFileSync, writeFileSync, existsSync, mkdirSync } from "fs";
import { dirname } from "path";

const MARKER_START = "<!-- git-skill:start -->";
const MARKER_END = "<!-- git-skill:end -->";

export function upsertSection(filePath: string, content: string): "created" | "updated" | "unchanged" {
  const snippet = `${MARKER_START}\n${content}\n${MARKER_END}`;

  if (!existsSync(filePath)) {
    mkdirSync(dirname(filePath), { recursive: true });
    writeFileSync(filePath, snippet + "\n");
    return "created";
  }

  const existing = readFileSync(filePath, "utf-8");
  const startIdx = existing.indexOf(MARKER_START);
  const endIdx = existing.indexOf(MARKER_END);

  if (startIdx !== -1 && endIdx !== -1) {
    const currentSection = existing.slice(startIdx, endIdx + MARKER_END.length);
    if (currentSection === snippet) return "unchanged";
    const updated = existing.slice(0, startIdx) + snippet + existing.slice(endIdx + MARKER_END.length);
    writeFileSync(filePath, updated);
    return "updated";
  }

  const separator = existing.endsWith("\n") ? "\n" : "\n\n";
  writeFileSync(filePath, existing + separator + snippet + "\n");
  return "updated";
}

export function removeSection(filePath: string): "removed" | "not_found" {
  if (!existsSync(filePath)) return "not_found";
  const content = readFileSync(filePath, "utf-8");
  const startIdx = content.indexOf(MARKER_START);
  const endIdx = content.indexOf(MARKER_END);
  if (startIdx === -1 || endIdx === -1) return "not_found";

  const before = content.slice(0, startIdx).replace(/\n+$/, "");
  const after = content.slice(endIdx + MARKER_END.length).replace(/^\n+/, "");
  const result = before + (after ? "\n\n" + after : "") + "\n";
  writeFileSync(filePath, result);
  return "removed";
}

// src/util/hooks.ts
import { readFileSync, writeFileSync, existsSync, chmodSync, unlinkSync } from "fs";
import { join } from "path";

const HOOK_MARKER = "# git-skill hook";
const HOOK_CONTENT = `\n${HOOK_MARKER}\ngit-skill capture --hook 2>/dev/null &\n`;

export function installHook(gitDir: string): "installed" | "updated" | "already_installed" {
  const hookPath = join(gitDir, "hooks", "post-commit");

  if (existsSync(hookPath)) {
    const content = readFileSync(hookPath, "utf-8");
    if (content.includes(HOOK_MARKER)) return "already_installed";
    writeFileSync(hookPath, content + HOOK_CONTENT);
    chmodSync(hookPath, 0o755);
    return "updated";
  }

  writeFileSync(hookPath, `#!/bin/sh\n${HOOK_CONTENT}`);
  chmodSync(hookPath, 0o755);
  return "installed";
}

export function removeHook(gitDir: string): "removed" | "not_found" {
  const hookPath = join(gitDir, "hooks", "post-commit");
  if (!existsSync(hookPath)) return "not_found";

  const content = readFileSync(hookPath, "utf-8");
  if (!content.includes(HOOK_MARKER)) return "not_found";

  const lines = content.split("\n");
  const filtered = lines.filter((line, i) => {
    if (line.includes(HOOK_MARKER)) return false;
    if (line.includes("git-skill capture")) return false;
    return true;
  });

  const result = filtered.join("\n").trim();
  if (result === "#!/bin/sh" || result === "") {
    unlinkSync(hookPath);
  } else {
    writeFileSync(hookPath, result + "\n");
  }
  return "removed";
}

export function hasHook(gitDir: string): boolean {
  const hookPath = join(gitDir, "hooks", "post-commit");
  if (!existsSync(hookPath)) return false;
  return readFileSync(hookPath, "utf-8").includes(HOOK_MARKER);
}

    .description("Capture the latest commit (called by post-commit hook)")
    .option("--hook", "Called from post-commit hook (suppress output)")
    .option("--hash <hash>", "Specific commit hash to capture")
    .action((opts: { hook?: boolean; hash?: string }) => {
      const cwd = process.cwd();
      const historyDir = join(cwd, ".git-history");

      if (!existsSync(historyDir)) {
        if (!opts.hook) write("No .git-history/ found. Run `git-skill init` first.\n");
        return;
      }

      const db = openDb(historyDir);

      try {
        const hash = opts.hash || getLastCommitHash(cwd);
        if (!hash) {
          if (!opts.hook) write("Could not determine last commit hash.\n");
          return;
        }

        // Check if already captured
        const existing = db.prepare("SELECT hash FROM commits WHERE hash = ?").get(hash);
        if (existing) {
          if (!opts.hook) write(`Commit ${hash.slice(0, 7)} already captured.\n`);
          return;
        }

        // Get commit data
        const commits = getLog(cwd, { limit: 1 });
        if (commits.length === 0) return;

        const commit = commits[0];
        if (commit.hash !== hash) {
          // If hash doesn't match latest, fetch specific commit
          const specific = getLog(cwd, { limit: 1, branch: hash });
          if (specific.length === 0) return;
          Object.assign(commit, specific[0]);
        }

        // Insert commit
        db.prepare(`
          INSERT OR IGNORE INTO commits (hash, message, author, email, timestamp, branch, parent_hash, merge_commit, insertions, deletions, files_changed)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        `).run(
          commit.hash, commit.message, commit.author, commit.email,
          commit.timestamp, commit.branch, commit.parentHash,
          commit.mergeCommit ? 1 : 0,
          commit.insertions, commit.deletions, commit.filesChanged
        );

        // Insert commit files
        const files = getDiffTree(cwd, commit.hash);
        const insertFile = db.prepare(`
          INSERT INTO commit_files (commit_hash, file_path, status, insertions, deletions, old_path)
          VALUES (?, ?, ?, ?, ?, ?)
        `);

        const insertMany = db.transaction((fileList: typeof files) => {
          for (const f of fileList) {
            insertFile.run(commit.hash, f.path, f.status, f.insertions, f.deletions, f.oldPath);
          }
        });
        insertMany(files);

        // Queue for embedding if enabled
        // (Embedding config check deferred to Plan 3)

        if (!opts.hook) {
          write(`Captured ${commit.hash.slice(0, 7)}: ${commit.message} (${files.length} files)\n`);
        }
      } finally {
        db.close();
      }
    });
}

// src/commands/snapshot.ts
import { Command } from "commander";
import { openDb } from "../util/db.js";
import { getLog, getDiffTree, getBranches, getTags } from "../util/git.js";
import { join } from "path";
import { existsSync, mkdirSync } from "fs";

function write(msg: string): void {
  process.stdout.write(msg);
}

export function snapshotCommand(): Command {
  return new Command("snapshot")
    .description("Full re-index — rebuild git history cache")
    .option("--force", "Force full rebuild (clear existing data)")
    .action((opts: { force?: boolean }) => {
      const cwd = process.cwd();
      const historyDir = join(cwd, ".git-history");

      mkdirSync(historyDir, { recursive: true });
      const db = openDb(historyDir);

      try {
        // Phase 1: Backfill commits
        write("Phase 1: Backfilling commits...\n");
        const existingHashes = new Set(
          (db.prepare("SELECT hash FROM commits").all() as { hash: string }[]).map(r => r.hash)
        );

        if (opts.force) {
          db.exec("DELETE FROM commits");
          db.exec("DELETE FROM commit_files");
          db.exec("DELETE FROM branches");
          db.exec("DELETE FROM tags");
          db.exec("DELETE FROM history_fts");
          existingHashes.clear();
          write("  Cleared existing data.\n");
        }

        const allCommits = getLog(cwd);
        const newCommits = allCommits.filter(c => !existingHashes.has(c.hash));

        write(`  Found ${allCommits.length} total commits, ${newCommits.length} new.\n`);

        const insertCommit = db.prepare(`
          INSERT OR IGNORE INTO commits (hash, message, author, email, timestamp, branch, parent_hash, merge_commit, insertions, deletions, files_changed)
          VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        `);

        const insertFile = db.prepare(`
          INSERT INTO commit_files (commit_hash, file_path, status, insertions, deletions, old_path)
          VALUES (?, ?, ?, ?, ?, ?)
        `);

        const backfill = db.transaction(() => {
          for (const commit of newCommits) {
            insertCommit.run(
              commit.hash, commit.message, commit.author, commit.email,
              commit.timestamp, commit.branch, commit.parentHash,
              commit.mergeCommit ? 1 : 0,
              commit.insertions, commit.deletions, commit.filesChanged
            );

            const files = getDiffTree(cwd, commit.hash);
            for (const f of files) {
              insertFile.run(commit.hash, f.path, f.status, f.insertions, f.deletions, f.oldPath);
            }
          }
        });
        backfill();

        write(`  Indexed ${newCommits.length} commits.\n`);

        // Phase 2: Branches
        write("Phase 2: Indexing branches...\n");
        db.exec("DELETE FROM branches");
        const branches = getBranches(cwd);
        const insertBranch = db.prepare(`
          INSERT OR REPLACE INTO branches (name, head_hash, created_at, is_active)
          VALUES (?, ?, ?, ?)
        `);
        for (const b of branches) {
          insertBranch.run(b.name, b.headHash, "", b.isActive ? 1 : 0);
        }
        write(`  Indexed ${branches.length} branches.\n`);

        // Phase 3: Tags
        write("Phase 3: Indexing tags...\n");
        db.exec("DELETE FROM tags");
        const tags = getTags(cwd);
        const insertTag = db.prepare(`
          INSERT OR REPLACE INTO tags (name, hash, timestamp, message)
          VALUES (?, ?, ?, ?)
        `);
        for (const t of tags) {
          insertTag.run(t.name, t.hash, t.timestamp, t.message);
        }
        write(`  Indexed ${tags.length} tags.\n`);

        // Phase 4: FTS rebuild
        write("Phase 4: Rebuilding FTS index...\n");
        db.exec("DELETE FROM history_fts");
        const allCommitsDb = db.prepare("SELECT hash, message FROM commits").all() as { hash: string; message: string }[];
        const insertFts = db.prepare("INSERT INTO history_fts (hash, type, path, message, detail) VALUES (?, ?, ?, ?, ?)");
        const ftsTransaction = db.transaction(() => {
          for (const c of allCommitsDb) {
            insertFts.run(c.hash, "commit", "", c.message, "");
          }
          const allFiles = db.prepare("SELECT commit_hash, file_path, status FROM commit_files").all() as { commit_hash: string; file_path: string; status: string }[];
          for (const f of allFiles) {
            insertFts.run(f.commit_hash, "file", f.file_path, "", f.status);
          }
        });
        ftsTransaction();
        write(`  FTS index built (${allCommitsDb.length} commits).\n`);

        // Update metadata
        db.prepare("INSERT OR REPLACE INTO schema_meta (key, value) VALUES ('last_snapshot', ?)").run(new Date().toISOString());

        write("Snapshot complete.\n");
      } finally {
        db.close();
      }
    });
}

// src/commands/init.ts
import { Command } from "commander";
import { join, dirname } from "path";
import { fileURLToPath } from "url";
import { existsSync, readFileSync, writeFileSync, appendFileSync } from "fs";
import { installHook } from "../util/hooks.js";
import { isGitRepo } from "../util/git.js";
import { execSync } from "child_process";

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

function write(msg: string): void {
  process.stdout.write(msg);
}

export function initCommand(): Command {
  return new Command("init")
    .description("Initialize git-skill in the current repository")
    .option("--skip-snapshot", "Skip initial snapshot")
    .option("--skip-cron", "Skip cron setup")
    .action((opts: { skipSnapshot?: boolean; skipCron?: boolean }) => {
      const cwd = process.cwd();

      if (!isGitRepo(cwd)) {
        write("Error: Not a git repository. Run from a git project root.\n");
        process.exit(1);
      }

      write("Initializing git-skill...\n\n");

      // Step 1: Install post-commit hook
      write("1. Installing post-commit hook...\n");
      const gitDir = join(cwd, ".git");
      const hookResult = installHook(gitDir);
      write(`   Hook: ${hookResult}\n`);

      // Step 2: Update .gitignore
      write("2. Updating .gitignore...\n");
      const gitignorePath = join(cwd, ".gitignore");
      const entriesToAdd = [".git-history/"];
      if (existsSync(gitignorePath)) {
        const content = readFileSync(gitignorePath, "utf-8");
        const missing = entriesToAdd.filter(e => !content.includes(e));
        if (missing.length > 0) {
          const suffix = content.endsWith("\n") ? "" : "\n";
          appendFileSync(gitignorePath, suffix + missing.join("\n") + "\n");
          write(`   Added ${missing.length} entries.\n`);
        } else {
          write("   Already up to date.\n");
        }
      } else {
        writeFileSync(gitignorePath, entriesToAdd.join("\n") + "\n");
        write("   Created .gitignore.\n");
      }

      // Step 3: Run initial snapshot
      if (!opts.skipSnapshot) {
        write("3. Running initial snapshot...\n");
        try {
          execSync(`node ${join(__dirname, "..", "index.js")} snapshot`, {
            cwd,
            encoding: "utf-8",
            stdio: "inherit",
          });
        } catch {
          // Try via process.argv[1] as fallback
          try {
            const entryPoint = process.argv[1];
            if (entryPoint) {
              execSync(`node ${entryPoint} snapshot`, { cwd, encoding: "utf-8", stdio: "inherit" });
            }
          } catch {
            write("   Warning: Could not run snapshot automatically. Run `git-skill snapshot` manually.\n");
          }
        }
      } else {
        write("3. Skipping snapshot.\n");
      }

      // Step 4: Cron (deferred to Plan 3)
      if (!opts.skipCron) {
        write("4. Cron setup deferred — run `git-skill cron` later.\n");
      } else {
        write("4. Skipping cron.\n");
      }

      write("\ngit-skill initialized successfully!\n");
      write("Post-commit hook will capture commits automatically.\n");
      write("Run `git-skill doctor` to verify setup.\n");
    });
}

// src/commands/doctor.ts
import { Command } from "commander";
import { join } from "path";
import { existsSync, readFileSync } from "fs";
import { isGitRepo } from "../util/git.js";
import { hasHook } from "../util/hooks.js";
import { hasDb, openDb } from "../util/db.js";

function write(msg: string): void {
  process.stdout.write(msg);
}

interface Check {
  name: string;
  status: "pass" | "warn" | "fail";
  message: string;
}

export function doctorCommand(): Command {
  return new Command("doctor")
    .description("Health check — verify git-skill setup")
    .option("--json", "Output as JSON")
    .action((opts: { json?: boolean }) => {
      const cwd = process.cwd();
      const checks: Check[] = [];

      // Check 1: Git repo
      const isRepo = isGitRepo(cwd);
      checks.push({
        name: "git repo",
        status: isRepo ? "pass" : "fail",
        message: isRepo ? "Git repository detected" : "Not a git repository",
      });

      // Check 2: Post-commit hook
      const gitDir = join(cwd, ".git");
      const hookInstalled = isRepo && hasHook(gitDir);
      checks.push({
        name: "post-commit hook",
        status: hookInstalled ? "pass" : "warn",
        message: hookInstalled ? "Hook installed" : "Hook not installed — run `git-skill init`",
      });

      // Check 3: History database
      const historyDir = join(cwd, ".git-history");
      const dbExists = hasDb(historyDir);
      checks.push({
        name: "history.db",
        status: dbExists ? "pass" : "fail",
        message: dbExists ? "Database exists" : "No database — run `git-skill snapshot`",
      });

      // Check 4: Snapshot freshness
      if (dbExists) {
        const db = openDb(historyDir);
        const meta = db.prepare("SELECT value FROM schema_meta WHERE key = 'last_snapshot'").get() as { value: string } | undefined;
        if (meta) {
          const lastSnapshot = new Date(meta.value);
          const hoursAgo = (Date.now() - lastSnapshot.getTime()) / 3600000;
          const fresh = hoursAgo < 24;
          checks.push({
            name: "snapshot freshness",
            status: fresh ? "pass" : "warn",
            message: fresh ? `Last snapshot: ${Math.round(hoursAgo)}h ago` : `Stale snapshot: ${Math.round(hoursAgo)}h ago — run \`git-skill snapshot\``,
          });

          const commitCount = (db.prepare("SELECT COUNT(*) as c FROM commits").get() as any).c;
          checks.push({
            name: "commit count",
            status: "pass",
            message: `${commitCount} commits indexed`,
          });
        }
        db.close();
      }

      // Check 5: .gitignore
      const gitignorePath = join(cwd, ".gitignore");
      if (existsSync(gitignorePath)) {
        const content = readFileSync(gitignorePath, "utf-8");
        const hasEntry = content.includes(".git-history");
        checks.push({
          name: ".gitignore",
          status: hasEntry ? "pass" : "warn",
          message: hasEntry ? ".git-history/ in .gitignore" : ".git-history/ not in .gitignore",
        });
      }

      // Output
      if (opts.json) {
        write(JSON.stringify({ checks }, null, 2) + "\n");
        return;
      }

      write("\ngit-skill doctor\n");
      write("─".repeat(40) + "\n");
      for (const check of checks) {
        const icon = check.status === "pass" ? "✓" : check.status === "warn" ? "!" : "✗";
        write(`  ${icon} ${check.name}: ${check.message}\n`);
      }
      write("\n");

      const failures = checks.filter(c => c.status === "fail");
      if (failures.length > 0) {
        process.exit(1);
      }
    });
}
